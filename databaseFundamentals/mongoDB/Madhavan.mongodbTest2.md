-- QUESTION 7
------------------------------------------------------
-- Find total spending by each customer.

db.orders.aggregate(
  [
    {$unwind:"$items"},
    {$lookup:{
      from:"customers",
      localField:"customerId",
      foreignField:"customerId",
      as:"customer"
    }},
    {
      $unwind:"$customer"
    },
    {
      $lookup:{
        from:"products",
        localField:"items.productId",
        foreignField:"productId",
        as:"product"
      }
    },
    {$unwind:"$product"},
    {
      $group:{
        _id:"$customer.customerName",
        totalSpending:{$sum:{$multiply:["$items.quantity","$product.price"]}}
      }
    },
    {
      $project:{_id:0,customerName:"$_id",totalSpending:1}
    }
  ]
)
